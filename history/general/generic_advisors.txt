# Ideological Political advisors
every_possible_country = {
	limit = {
		is_dynamic_country = no
		NOT = { OR = { original_tag = GER original_tag = ENG original_tag = RUS original_tag = FRA original_tag = ITA original_tag = JAP original_tag = USA original_tag = BEL original_tag = UKR original_tag = HUN original_tag = AUS  } }
		check_has_focus_tree_to_switch_to_communism = yes
		# avoid generic advisors in temporary countries
		is_dynamic_country = no
		NOT = {
			has_allowed_idea_with_traits = { idea = communist_revolutionary characters = yes }
		}
	}
	generate_character = {
		token_base = generic_communist_revolutionary # token will be TAG_token_base
		advisor = {
			slot = political_advisor
			traits = { communist_revolutionary }
			available = {
			}
			visible = {
				OR = {
					date > 1917.1.1
					has_country_flag = great_war_loser
					has_completed_focus = socialist_reforms
				}
			}
			ai_will_do = {
				factor = 0
			}
		}
		portraits = { 
			army = { small = GFX_idea_generic_communist_revolutionary }
		}
	}
	generic_communist_revolutionary = { set_character_flag = is_generic }
}
