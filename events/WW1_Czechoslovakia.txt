				owns_state = 993
			}
		}
	}
	option = {
		name = ww1_czechoslovakia.1.a
		every_state = {
			limit = {
				OR = {
					is_owned_by = AUR
					owner = { has_country_flag = great_war_loser }
				}
				OR = {
					state = 69
					state = 74
					state = 992
					state = 993
					state = 995
				}
			}
			set_demilitarized_zone = no
			transfer_state_to = CZE
		}
		every_state = {
			limit = {
				is_owned_by = CZE
				NOT = {
					state = 69
					state = 74
					state = 992
					state = 993
					state = 995
				}
			}
			remove_core_of = AUS
		}
		AUR = {
			country_event = ww1_czechoslovakia.3
		}

			factor = 100
		}
	}
