			limit = {

					state = 995

		}
		every_state = {
			limit = {
				is_owned_by = CZE

					state = 69
					state = 1200
					state = 74
					state = 1199

					state = 995
				}
			}

	fire_only_once = yes
	mean_time_to_happen = {
		days = 0
	}
	option = {
		name = ww1_czechoslovakia.6.a
		effect_tooltip = {
			SLO = { change_tag_from = ROOT }
		}
		hidden_effect = {
			if = {
				limit = {
					is_subject = yes
				}
				SLO = { copy_autonomy_from_ROOT = yes }
			}
			SLO = {
				take_all_subjects_ROOT = yes
				copy_government_of_ROOT = yes

				}
				else = {
					custom_effect_tooltip = GAME_OVER_TT
				}
			}
		}
	}
}
