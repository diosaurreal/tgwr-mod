	id = ww1_montenegro.4
	title = ww1_montenegro.4.t
	desc = ww1_montenegro.4.d
	picture = GFX_report_event_generic_parliament
	trigger = {
		tag = MNT
		has_war = no
		has_country_leader = {
			character = MNT_nikola_i
			ruling_only = yes
		}
		has_global_flag = great_war_ends
		OR = {
			has_country_flag = great_war_loser
			AND = {
				has_country_flag = Montenegro_capitulated
				has_country_flag = great_war_winner
			}
		}
		is_subject = no
	}
	fire_only_once = yes
	mean_time_to_happen = {
		days = 15
	}
	option = {
		name = ww1_montenegro.4.a
		retire_character = MNT_nikola_i
		set_politics = {
			ruling_party = democratic
			elections_allowed = no
		}
		set_party_name = {
			ideology = democratic
			long_name = MNT_national_assembly_long
			name = MNT_national_assembly
		}
		MNT_podgorica_assembly = {
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					ideology = liberal_conservatism
				}
			}
		}
		hidden_effect = {
			MNT_danilo_petrovic_njegos = {
				remove_unit_leader_role = yes
				promote_character = semi_constitutional_monarchy
				set_character_name = {
					name = MNT_danilo_ii
				}
			}
		}
	}
}

	id = ww1_montenegro.6
	title = ww1_montenegro.6.t
	desc = ww1_montenegro.6.d
	is_triggered_only = yes
	trigger = {
		owns_state = 105
	}
	fire_only_once = yes
	immediate = {
		hidden_effect = {
			MNT = {
				MNT_vukasin_markovic = {
					promote_character = vanguardism
				}
				MNT_krsto_popovic = {
					promote_character = stratocracy
				}
				MNT_jovan_plamenac = {
					promote_character = authoritarian_democracy
				}
				set_politics = {
					ruling_party = fascism
					elections_allowed = no
				}
				set_popularities = {
					democratic = 12
					fascism = 40
					neutrality = 48
				}
				set_party_name = {
					ideology = fascism
					long_name = MNT_zeleni_party_long
					name = MNT_zeleni_party
				}
			}
		}
	}
	option = {
		name = ww1_montenegro.6.a
		105 = {
			damage_building = {
				type = industrial_complex
				damage = 0.5
			}
		}
		add_manpower = -30
		MNT = { add_manpower = -98 }
	}
	option = { #The Greens
		name = ww1_montenegro.6.b
		MNT = {
			transfer_state = 105
			transfer_state = 1205
			hidden_effect = {
				division_template = {
					name = "Zelenaši"
					regiments = {
						infantry = { x = 0 y = 0 }
						infantry = { x = 0 y = 1 }
						infantry = { x = 0 y = 2 }
					}
				}
				capital_scope = {
					create_unit = {
						division = "name = \"Zelenaši\" division_template = \"Zelenaši\" start_experience_factor = 1.0 start_equipment_factor = 0.7"
						owner = PREV
						count = 2
					}
				}
			}
			declare_war_on = {
				target = ROOT
				type = independence_wargoal
			}
		}
		ai_chance = {
			base = 0
		}
	}
}
