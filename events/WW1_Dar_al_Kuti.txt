add_namespace = ww1_kuti

# Muhammad al-Sanussi dies, game over
country_event = {
	id = ww1_kuti.1
	title = ww1_kuti.1.t
	desc = ww1_kuti.1.d
	
	picture = GFX_report_event_african_soldiers
	
	fire_only_once = yes
	mean_time_to_happen = { days = 12 }
	trigger = {
		tag = KUT
		date > 1911.1.1
		has_start_date < 1914.1.1
		has_character = KUT_muhammad_al_sanussi
	}
	
	option = {
		name = ww1_kuti.1.a
		custom_effect_tooltip = GAME_OVER_TT
		retire_character = KUT_muhammad_al_sanussi
		promote_character = {
			character = KUT_kamun_al_sanussi
			ideology = theocracy_ideology
		}	
		promote_character = {
			character = KUT_kamun_al_sanussi
			ideology = fundamentalism
		}
		clr_country_flag = protectorate
		if = {
			limit = {
				is_subject_of = FRA
			}
			OVERLORD = {
				if = {
					limit = {
						has_subject = AEF
					}
					AEF = {
						annex_country = {
							target = ROOT
							transfer_troops = no
						}								
					}
					else = {
						annex_country = {
							target = ROOT
							transfer_troops = no
						}
					}
				}
				country_event = ww1_kuti.2
			}
		}
		1142 = {
			if = {
				limit = { has_resistance = no }
				start_resistance = yes			
			}	
			add_resistance = 20
			add_dynamic_modifier = {
				modifier = FRA_anti_french_resistance_mid
			}
		}
	}
	option = {
		name = ww1_kuti.1.b
		retire_character = KUT_muhammad_al_sanussi
		promote_character = {
			character = KUT_kamun_al_sanussi
			ideology = theocracy_ideology
		}	
		promote_character = {
			character = KUT_kamun_al_sanussi
			ideology = fundamentalism
		}
		trigger = {
			is_subject_of = FRA
			FRA = {
				has_war_with = CHA
			}
		}
		set_temp_variable = { target = FRA }
		clear_all_relations_with_TARGET = yes
		add_to_war = {
			targeted_alliance = CHA
			enemy = FRA
		}
		ai_chance = {
			base = 0
		}
	}
}
