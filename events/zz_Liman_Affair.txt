	fire_only_once = yes

		add_war_support = -0.1

			FROM = {
				original_tag = TUR
			}
		}
		TUR = {
			country_event = {
				id = ww1_liman_affair.2 #Turkey is informed on Germany's refusal
				days = 2
			}
		}
		ai_chance = {
			factor = 0.25
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
			modifier = {
				NOT = {
					is_historical_focus_on = yes
				}
				GER = {
					has_opinion = {
						target = TUR
						value > 0
					}
				}
				factor = 1.5
			}
		}
	}
}
