country_event = {
	id = ww1_feudal.1
	title = ww1_feudal.1.t
	desc = ww1_feudal.1.d
	picture = GFX_report_event_royal_parade
	is_triggered_only = yes
	option = {
		name = ww1_feudal.1.a
		if = {
			limit = {
				FROM = {
					NOT = {
						original_tag = JOH
					}
				}
			}
			FROM = {
				set_country_flag = protectorate
				set_country_flag = ENG_accepts_protection_FEU
				country_event = ww1_feudal.2
			}
			set_autonomy = {
				target = FROM
				autonomy_state = autonomy_tgwr_informal_protectorate
			}
			restore_previous_government_type = yes
		}
		else = {
			set_country_flag = can_take_malay_focus_ENG
			custom_effect_tooltip = unlocks_malay_focus_ENG
		}
		if = {
			limit = {
				FROM = {
					any_owned_state = {
						is_claimed_by = ENG
						is_core_of = FROM
					}
				}
			}
			every_state = {
				limit = {
					is_core_of = FROM
					is_claimed_by = ENG
				}
				remove_claim_by = ENG
			}
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 100
				FROM = {
					OR = {
						original_tag = JOH
						original_tag = QAT
					}
				}
			}
		}
	}
	option = {
		name = ww1_feudal.1.b
		FROM = {
			country_event = ww1_feudal.3
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 100
				FROM = {
					original_tag = BID
				}
			}
		}
	}
}
