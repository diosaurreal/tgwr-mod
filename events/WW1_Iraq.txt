# Kingdom of Iraq
country_event = {
	id = ww1_iraq.1
	title = ww1_iraq.1.t
	desc = ww1_iraq.1.d
	picture = GFX_report_event_tur_turkish_soldiers
	fire_only_once = yes
	mean_time_to_happen = {
		days = 23
	}
	trigger = {
		tag = ENG
		is_subject = no
		owns_state = 291
		controls_state = 291
		NOT = {
			country_exists = IRQ
		}
		SYR = {
			OR = {
				exists = no
				NOT = {
					has_country_leader = {
						character = HJZ_faisal_bin_hussein_bin_ali
					}
				}	
			}
		}
		date > 1921.8.1
		has_global_flag = interwar_iraqi_revolt
	}
	option = {
		name = ww1_iraq.1.a
		set_global_flag = kingdom_of_iraq
		release_puppet = IRQ
		IRQ = { set_country_flag = protectorate }
		set_autonomy = {
			target = IRQ
			autonomy_state = autonomy_tgwr_protectorate
			freedom_level = 0.8
		}
		HJZ = {
			HJZ_faisal_bin_hussein_bin_ali = {
				set_nationality = IRQ
				promote_leader = yes
				set_character_name = SYR_faisal_i
				if = {
					limit = {
						has_ideology = semi_constitutional_monarchy
					}
					promote_character = semi_constitutional_monarchy
				}
				else = {
					add_country_leader_role = {
						promote_leader = yes
						country_leader = {
							ideology = semi_constitutional_monarchy
						}
					}
				}
			}
			HJZ_ghazi_bin_faisal = {
				set_nationality = IRQ
			}
		}
		hidden_effect = {
			IRQ = {
				set_politics = {
					ruling_party = neutrality
					elections_allowed = no
				}
				set_popularities = {
					democratic = 18
					fascism = 1
					communism = 1
					neutrality = 80
				}
			}
			every_state = {
				limit = {
					is_core_of = IRQ
					is_claimed_by = ROOT
				}
				remove_claim_by = ROOT
			}
		}
	}
	option = {
		name = ww1_iraq.1.b
		release = IRQ
		declare_war_on = {
			target = IRQ
			type = annex_everything
		}
		ai_chance = {
			factor = 0
		}
	}
}
