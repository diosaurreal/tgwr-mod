			limit = { has_idea = TRS_pro_ottoman_opposition_low }
			swap_ideas = {

			}
		}
		else_if = {
			limit = { has_idea = TRS_pro_ottoman_opposition_medium }
			swap_ideas = {
				remove_idea = TRS_pro_ottoman_opposition_medium
				add_idea = TRS_pro_ottoman_opposition_high
			}
		}
		else_if = {
			limit = { has_idea = TRS_pro_ottoman_opposition_high }
			swap_ideas = {
				remove_idea = TRS_pro_ottoman_opposition_high
