	# Treaty of Rapallo is signed
	option = {
		name =  ww1_german_interwar.17.a
		add_ideas = GER_idea_treaty_of_rapallo
		add_opinion_modifier = {
			target = FROM
			modifier = medium_increase
		}
		FROM = {
			add_ideas = GER_idea_treaty_of_rapallo
			add_opinion_modifier = {
				target = PREV
				modifier = medium_increase
			}
		}
		every_state = {
			limit = {
				OR = {
					AND = {
						is_owned_by = POL
						POL = {
							NOT = {
								has_government = communism
								is_subject_of = SOV
							}
						}
