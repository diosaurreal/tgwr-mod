# Faisal bin Turki Dies!
country_event = {
	title = ww1_oman.1.t
	desc = ww1_oman.1.d
	id = ww1_oman.1
	picture = GFX_report_event_royal_parade
	
	fire_only_once = yes
	mean_time_to_happen = { days = 9 }
	trigger = {
		original_tag = MSC
		has_character = OMA_faisal_bin_turki
		date > 1913.10.1
		has_start_date < 1914.1.1
		has_country_leader = { character = OMA_faisal_bin_turki ruling_only = no }
	}
	
	option = {
		name = ww1_oman.1.a
		retire_character = OMA_faisal_bin_turki
		promote_character = OMA_taimur_bin_feisal
	}
}

		trigger = {
			NOT = {
				has_war_with = FROM
			}
		}
		name = ww1_oman.2.a
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				has_opinion = {
					target = FROM
					value < 0
				}
				NOT = {
					has_country_flag = great_war_loser
					has_idea = idea_armistice
					has_idea = idea_army_restrictions
				}
			}
			modifier = {
				factor = 0
				strength_ratio = {
					tag = FROM
					ratio > 2
				}
				NOT = {
					has_country_flag = great_war_loser
					has_idea = idea_armistice
					has_idea = idea_army_restrictions
				}
			}
		}
		every_owned_state = {
			limit = {
				OR = {
					is_core_of = SOM
					is_core_of = HBY
					is_core_of = MJT
					is_core_of = GLD
				}
			}
			transfer_state_to = FROM
		}
	}
	option = {
		name = ww1_oman.2.b
		trigger = {
			NOT = {
				has_country_flag = great_war_loser
				has_idea = idea_armistice
				has_idea = idea_army_restrictions
			}
		}
		ai_chance = { base = 100 }
		FROM = {
			create_wargoal = {
				target = ROOT
				type = puppet_wargoal_focus
			}
		}
	}
}

# Buy Zanzibar
country_event = {
	id = ww1_oman.3
	title = ww1_oman.3.t
	desc = ww1_oman.3.d
	picture = GFX_report_event_generic_sign_treaty1

	is_triggered_only = yes
	
	option = {
		name = ww1_oman.3.a
		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_opinion = {
					target = FROM
					value < 0
				}
			}
		}
		if = {
			limit = {
				has_subject = ZAN
			}
			ZAN = {
				get_current_government_type = yes
				FROM = {
					set_autonomy = {
						target = PREV
						autonomy_state = autonomy_dominion
					}
				}
				restore_previous_government_type = yes
			}
		}
		else = {
			FROM = {
				transfer_state = 903
			}
		}
		add_timed_idea = {
			idea = MSC_Zanzibar_money
			days = 70
		}
	}
	option = {
		name = ww1_oman.3.b
		ai_chance = { base = 75 }
		FROM = {
			add_opinion_modifier = {
				target = ROOT
				modifier = large_decrease
			}
		}
	}
}

	picture = GFX_report_event_gathering_protest
	
	fire_only_once = yes
	mean_time_to_happen = { days = 30 }
	trigger = {
		tag = MSC
		owns_state = 882
		date > 1913.5.1
		has_start_date < 1914.1.1
	}
	
	option = {
		name = ww1_oman.5.a
		882 = {
			remove_core_of = MSC
			add_claim_by = MSC
			add_core_of = OMA
			start_resistance = yes
			if = {
				limit = { ROOT = { has_war = no } }
				set_demilitarized_zone = yes
			}
		}
		hidden_effect = {
			OMA = {
				inherit_technology = ROOT
			}
		}
	}
}
