# Stephan Burian Resigns from Position
country_event = {
	id = ww1_austriahungary.1
	title = ww1_austriahungary.1.t
	desc = ww1_austriahungary.1.d
	#picture = GFX_report_event_generic_read_write
	fire_only_once = yes
	trigger = {
		#has habsburg
		#hungary expells the ban of Ban of Croatia-Slavonia and establishes a dictatorship there
		has_government = neutrality
		date > 1912.02.14 #feb 1912 he left the post
	}
	option = {
		name = ww1_austriahungary.1.a
		AUS_stephan_burian = {
			set_character_flag = stephan_burian_break
		}
	}
}

#Stephan Burian appointed Minister besides the King
country_event = {
	id = ww1_austriahungary.2
	title = ww1_austriahungary.2.t
	desc = ww1_austriahungary.2.d
	#picture = GFX_report_event_generic_read_write
	fire_only_once = yes
	trigger = {
		#has habsburg
		has_government = neutrality
		date > 1913.06.01
	}
	option = {
		name = ww1_austriahungary.2.a
		remove_trait = {
		    character = AUS_stephan_burian
		    trait = reichsfinanzminister
		}
		add_trait = {
		    character = AUS_stephan_burian
		    trait = minister_besides_king
		}
		AUS_stephan_burian = {
			clr_character_flag = stephan_burian_break
		}
	}
}
