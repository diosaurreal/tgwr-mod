	id = ww1_austriahungary.1
	title = ww1_austriahungary.1.t
	desc = ww1_austriahungary.1.d
	#picture = GFX_report_event_generic_read_write
	fire_only_once = yes

		name = ww1_austriahungary.1.a

			set_character_flag = stephan_burian_break
		}
	}
}
