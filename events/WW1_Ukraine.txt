		tag = UKR

		has_global_flag = great_war_ends

			}
		}
	}
	picture = GFX_report_event_gathering_protest

	}

			limit = {
				is_in_faction_with = GER

			}
			leave_faction = yes
		}
		set_politics = {
			ruling_party = democratic

			character = UKR_volodymyr_vynnychenko

			limit = { has_idea = GER_idea_german_military_advisors }

			}

# Treaty of Warsaw (Polish-Ukrainian Agreement)
## Ukranian victory
country_event = {
	id = ww1_ukraine.13
	title = ww1_ukraine.13.t
	desc = {
		text = ww1_ukraine.13.d_sov
		trigger = {
			has_war_with = SOV
			NOT = {
				FROM = { has_war_with = SOV }
				has_global_flag = polish_soviet_war
			}
		}
	}
	desc = {
		text = ww1_ukraine.13.d
		trigger = {
			OR = {
				NOT = {
					has_war_with = SOV
				}
				has_global_flag = polish_soviet_war
				FROM = { has_war_with = SOV }
			}
		}
	}
	picture = GFX_report_event_generic_read_write
	is_triggered_only = yes
	option = {
		name = ww1_ukraine.13.a
		hidden_effect = {
			set_country_flag = treaty_of_warsaw
			WUK = {
				transfer_units_fraction= {
					target = UKR
					size = 1
					target_organization = 1
					source_organization = 1
					stockpile_ratio = 1
					army_ratio = 1
					navy_ratio = 1
					air_ratio = 1
				}
			}
		}
		FROM = {
			set_country_flag = treaty_of_warsaw
			every_owned_state = {
				limit = {
					OR = {
						is_core_of = ROOT
						is_claimed_by = ROOT
						state = 94
						state = 194
						state = 241
						state = 808
					}
					NOT = { state = 237 }
				}
				transfer_state_to = ROOT
			}
			every_country = {
				limit = {
					OR = {
						tag = PREV
						is_ally_with = PREV
					}
				}
				set_truce = {
					target = ROOT
					days = 365
				}
				white_peace = ROOT
			}
		}
		if = {
			limit = {
				country_exists = WUK
			}
			annex_country = {
				target = WUK
				transfer_troops = yes
			}
			hidden_effect = {
				WUK = {
					every_core_state = {
						add_core_of = ROOT
						add_core_of = UKR
						remove_core_of = WUK
					}
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_war_with = SOV
					is_ally_with = SOV
					has_government = communism
					has_global_flag = polish_soviet_war
				}
				FROM = { has_war_with = SOV }
			}
			FROM = {
				add_to_war = {
					targeted_alliance = ROOT
					enemy = SOV
				}
			}
			hidden_effect = {
				polish_soviet_war_effect = yes
			}
		}
	}
	option = {
		name = ww1_ukraine.13.b
		add_political_power = -100
		ai_chance = { factor = 0 }
	}
}
