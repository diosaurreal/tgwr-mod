add_namespace = ww1_tibet
#New Flag
country_event = {
	id = ww1_tibet.1
	title = ww1_tibet.1.t
	desc = ww1_tibet.1.d	
	picture = GFX_report_event_generic_read_write
	
	fire_only_once = yes

	trigger = {
		original_tag = TIB
		date > 1916.1.1
		has_government = neutrality
		NOT = {
			has_cosmetic_tag = TIB_1916
		}	
	}

	mean_time_to_happen = { days = 50 }
	
	option = {
		name = ww1_tibet.1.a
		set_cosmetic_tag = TIB_1916
	}
}
