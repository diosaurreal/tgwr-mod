	fire_only_once = yes
	mean_time_to_happen = {

	}

				retire_character = ICE_frederick_viii
				promote_character = ICE_christian_x
				remove_ideas = ICE_frederick_viii
				add_ideas = ICE_tgwr_christian_x
			}
		}
	}
}

# JUTLAND REFERENDUM
#######################
country_event = {
	id = ww1_denmark.3
	title = ww1_denmark.3.t
	desc = ww1_denmark.3.d
	picture = GFX_report_event_generic_peaceful_annexation
	fire_only_once = yes
	mean_time_to_happen = {
		days = 90
	}
	trigger = {
		original_tag = DEN
		any_other_country = {
			owns_state = 751
			has_country_flag = great_war_loser
			is_neighbor_of = PREV
			NOT = {
				has_war_with = PREV
				has_subject = PREV
			}
		}
	}
	immediate = {
		hidden_effect = {
			751 = {
				set_demilitarized_zone = yes
			}
		}
	}
	option = {
		name = ww1_denmark.3.a
		751 = {
			set_demilitarized_zone = no
			remove_core_of = GER
			remove_core_of = SHL
			transfer_state_to = PREV
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				has_country_flag = great_war_winner
			}
		}
	}
	option = {
		name = ww1_denmark.3.b
		trigger = {
			has_country_flag = great_war_winner
		}
		751 = {
			set_demilitarized_zone = no
			remove_core_of = GER
			remove_core_of = SHL
			transfer_state_to = PREV
		}
		58 = {
			transfer_state_to = PREV
		}
		ai_chance = {
			factor = 50
		}
	}
	option = {
		name = ww1_denmark.3.c
		trigger = {
			has_country_flag = great_war_winner
			NOT = { country_exists = SHL }
		}
		751 = {
			set_demilitarized_zone = no
			remove_core_of = GER
			remove_core_of = SHL
			transfer_state_to = PREV
		}
		58 = {
			transfer_state_to = SHL
		}
		if = {
			limit = { has_monarchy = yes }
			SHL = { set_country_flag = monarchist_puppet }
			set_autonomy = {
				target = SHL
				autonomy_state = autonomy_tgwr_personal_union
			}
		}
		else = {
			SHL = { set_country_flag = protectorate }
			set_autonomy = {
				target = SHL
				autonomy_state = autonomy_tgwr_protectorate
			}
		}
		ai_chance = {
			factor = 50
		}
	}
	option = {
		name = ww1_denmark.3.e
		trigger = {
			has_country_flag = great_war_loser
		}
		751 = {
			set_demilitarized_zone = no
		}
		58 = {
			owner = {
				transfer_state = 751
			}
		}
		remove_state_core = 751
		ai_chance = {
			factor = 200
		}
	}
}
