# Kurt Eisner Assassinated
country_event = {
	id = freikorps.1
	title = freikorps.1.t
	desc = freikorps.1.d
	picture = GFX_report_event_picture_kurt_eisner_death
	trigger = {
		OR = {
			AND = {
				tag = GER
				BAY = {
					exists = no
				}
			}
			tag = BAY
		}		
		NOT = {
			GER = {
				has_government = communism
			}
			# has_global_flag = spartakist_in_bavaria
			has_global_flag = prussian_blue
			AUS = {
				has_country_flag = great_war_winner
			}
		}
		GER = {
			has_country_flag = great_war_loser
		}
		BAY = {
			has_country_flag = great_war_loser
			has_country_leader = {
				character = BAY_kurt_eisner
				ruling_only = yes
			}
		}
	}
	fire_only_once = yes
	mean_time_to_happen = { days = 120 }
	option = {
		name = freikorps.1.a
		BAY = {
			retire_character = BAY_kurt_eisner
			promote_character = BAY_ernst_toller
			promote_character = BAY_johannes_hoffmann
			hidden_effect = {
				set_popularities = {
					fascism = 21
					communism = 44
					democratic = 33
					neutrality = 2
				}
				meta_effect = {
					text = {
						set_politics = {
							ruling_party = communism
							last_election = "[CURRENT_DATE]"
							elections_allowed = no
						}
					}
					CURRENT_DATE = "[?global.date.GetDateStringNoHour]"
				}
				set_cosmetic_tag = BAY_socialism
				add_ideas = standing_alone
			}
			transfer_state = 52
			send_embargo = GER
			set_truce = {
				target = GER
				days = 365
			}
		}
	}
}
