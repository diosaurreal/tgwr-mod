# Salomón–Lozano Treaty
country_event = {
	id = ww1_colombia.6
	title = ww1_colombia.6.t
	desc = ww1_colombia.6.d
	picture = GFX_report_event_generic_conference
	trigger = {
		tag = COL
		owns_state = 1086
		is_subject = no
		PRU = {
			owns_state = 1083
			is_neighbor_of = ROOT
		}
		date > 1922.3.24
	}
	mean_time_to_happen = {
		days = 10
	}
	fire_only_once = yes
	option = {
		name = ww1_colombia.6.a
		every_state = {
			limit = {
				OR = {
					state = 1084
					state = 1085
					state = 1087
				}
			}
			if = {
				limit = {
					is_owned_by = ROOT
				}
				transfer_state_to = PRU
			}
			remove_claim_by = ROOT
		}
		effect_tooltip = {
			PRU = {
				every_state = {
					limit = {
						OR = {
							state = 1083
							state = 1086
						}
					}
					if = {
						limit = {
							is_owned_by = PREV
						}
						transfer_state_to = ROOT
					}
					remove_claim_by = PREV
				}
			}
		}
		hidden_effect = {
			PRU = {
				country_event = ww1_colombia.7
			}
		}
	}
}
