# Andranik Returns
country_event = {
	id = ww1_armenia.2
	fire_only_once = yes
	trigger = {
		tag = ARM
		is_subject = no
		OR = {
			has_global_flag = great_war_ends
			has_global_flag = mudros_armistice
		}
		TUR = { has_country_flag = great_war_loser }
		any_other_country = {
			original_tag = ARM
			has_country_leader = {
				character = ARM_andranik_ozanian
			}
			NOT = {
				has_war_with = ROOT
			}
		}
	}
	hidden = yes
	mean_time_to_happen = { days = 10 }
	immediate = {
		random_other_country = {
			limit = {
				original_tag = ARM
				has_country_leader = {
					character = ARM_andranik_ozanian
				}
			}
			ARM_andranik_ozanian = {
				remove_country_leader_role = {
					ideology = revolutionary_nationalism
				}
				remove_country_leader_role = {
					ideology = authoritarian_democracy
				}
			}
			every_character = {
				set_nationality = ROOT
			}
			every_state = {
				limit = {
					OR = {
						is_core_of = PREV
						is_claimed_by = PREV
					}
				}
				remove_core_of = PREV
				remove_claim_by = PREV
			}
			delete_unit_template_and_units = {
				division_template = "Striking Division"
				disband = yes
			}
			ROOT = {
				annex_country = {
					target = PREV
					transfer_troops = yes
				}
			}
		}
	}
}

# Treaty of Alexandropol

	picture = GFX_report_event_armenian_resistance
	is_triggered_only = yes
	option = {
		name = ww1_armenia.3.a
		white_peace = FROM

			limit = {
				is_owned_by = FROM

					is_core_of = ROOT
					is_claimed_by = ROOT
				}
				NOT = {
					state = 230
					state = 787
					state = 939
				}
			}
			transfer_state_to = ROOT
		}
		if = {
			limit = {
				FROM = { owns_state = 818 }
				OR = {
					is_in_faction_with = AZR
					has_subject = AZR
					AZR = { has_war_with = FROM }

				}
			}
			AZR = { white_peace = FROM }

		set_global_flag = treaty_of_alexandropol

		}
	}
