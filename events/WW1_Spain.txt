# 1923 Spanish Coup
country_event = {
	id = ww1_spain.2
	title = ww1_spain.2.t
	desc = ww1_spain.2.d
	picture = GFX_report_event_spain_civil_war_soldiers
	fire_only_once = yes
	mean_time_to_happen = {	days = 13 }
	trigger = {
		tag = SPR
		has_character = SPR_miguel_primo_de_rivera
		has_country_flag = disaster_of_annual
		is_subject = no
		has_government = neutrality
		date > 1923.9.1
	}
	option = {
		name = ww1_spain.2.a
		hidden_effect = {
			set_country_flag = spanish_junta
			add_popularity = {
				ideology = fascism
				popularity = 0.2
			}
			add_popularity = {
				ideology = neutrality
				popularity = 0.2
			}
		}
		SPR_miguel_primo_de_rivera = {
			if = {
				limit = { 
					has_ideology = personalistic_dictatorship
				}
				promote_character = personalistic_dictatorship
			}
			else = {
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = personalistic_dictatorship
					}
				}
			}
			if = {
				limit = { is_field_marshal = no }
				promote_leader = yes
			}
		}	
		set_politics = {
			ruling_party = neutrality
			long_name = SPR_military_directory_party
			name = SPR_military_directory_party
			elections_allowed = no
		}
		add_ideas = SPR_alfonso_xiii
		add_stability = -0.10
	}
}

	picture = GFX_report_event_spain_civil_war_soldiers
	fire_only_once = yes
	mean_time_to_happen = {	days = 30 }
	trigger = {
		tag = SPR
		has_country_leader = {
			character = SPR_miguel_primo_de_rivera
		}
		SPR_miguel_primo_de_rivera = {
			has_ideology = personalistic_dictatorship
		}
		has_idea = SPR_alfonso_xiii
		date > 1925.12.1
	}
	option = {
		name = ww1_spain.3.a
		SPR_miguel_primo_de_rivera = {
			remove_country_leader_role = {
				ideology = personalistic_dictatorship
			}
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					ideology = civilian_dictatorship
				}
			}
		}
		set_party_name = {
			ideology = neutrality
			long_name = SPR_civil_directory_party
			name = SPR_civil_directory_party
		}
		add_popularity = {
			ideology = neutrality
			popularity = -0.05
		}
		add_popularity = {
			ideology = fascism
			popularity = -0.05
		}
		add_stability = -0.05
	}
}

				add_country_leader_role = {
					promote_leader = yes

						ideology = authoritarian_democracy
					}
				}
			}
		}
		add_stability = -0.10
