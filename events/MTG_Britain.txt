add_namespace = mtg_britain

# The Fate of the Royal Family event
country_event = {
	id = mtg_britain.43
	title = mtg_britain.43.t
	desc = {
		text = canada.69.c
		trigger = {
			any_enemy_country = {
				has_country_flag = has_british_monarchy
				has_capitulated = yes
			}
		}
	}
	desc = {
		text = mtg_britain.43.d
		trigger = {
			NOT = {
				any_other_country = {
					has_country_flag = has_british_monarchy
				}
			}
		}
	}
	picture = GFX_report_event_eng_royal_family

	trigger = {
		tag = ENG
		has_civil_war = no
		has_government = communism
		# No Where To Run
		NOT = {
			any_enemy_country = {
				original_tag = ROOT
			}
		}
		OR = {
			any_enemy_country = {
				has_country_flag = has_british_monarchy
				has_capitulated = yes
			}
			NOT = {
				any_other_country = {
					has_country_flag = has_british_monarchy
				}
			}
		}
	}
	fire_only_once = yes
	mean_time_to_happen = {
		days = 40
		modifier = {
			factor = 0.1
			any_enemy_country = {
				has_country_flag = has_british_monarchy
				has_capitulated = yes
			}
		}
	}
	
	# Banish them to Canada
	option = {
		name = mtg_britain.43.a
		trigger = {
			CAN = {
				exists = yes
				NOT = {
					has_government = communism
					has_country_flag = has_british_monarchy
				}
			}
			NOT = {
				any_enemy_country = {
					has_country_flag = has_british_monarchy
					has_capitulated = yes
				}
			}
		}
		ai_chance = {
			factor = 75
		}
		add_timed_idea = { idea = ENG_confiscated_royal_wealth days = 365 }
		CAN = {
			country_event = { id = mtg_britain.44 }
			effect_tooltip = {
				add_ideas = ENG_exiled_royal_family
			}
		}
	}

	# Execute them all
	option = {
		name = mtg_britain.43.b
		ai_chance = {
			factor = 25
		}
		add_timed_idea = { idea = ENG_confiscated_royal_wealth days = 365 }
		add_stability = -0.1
		if = {
			limit = {
				has_country_leader = {
					character = ENG_george_v
					ruling_only = no
				}
			}
			retire_character = ENG_george_v
		}
		else_if = {
			limit = {
				has_country_leader = {
					character = ENG_edward_viii
					ruling_only = no
				}
			}
			retire_character = ENG_edward_viii
		}
		else_if = {
			limit = {
				has_country_leader = {
					character = GER_carl_eduard_von_sachsen_coburg_und_gotha
					ruling_only = no
				}
			}
			retire_character = GER_carl_eduard_von_sachsen_coburg_und_gotha
		}
		hidden_effect = {
			if = {
				limit = {
					any_enemy_country = {
						has_country_flag = has_british_monarchy
						has_capitulated = yes
					}
				}
				every_other_country = {
					limit = {
						has_country_flag = has_british_monarchy
					}
					set_politics = {
						ruling_party = fascism
						elections_allowed = no
					}
					add_popularity = {
						ideology = fascism
						popularity = 0.2
					}
					remove_ideas = ENG_imperial_refuge
					news_event = { id = canada.78 days = 1 }
				}
			}
			else = {
				news_event = { id = mtg_news.35 days = 1 }
			}
		}
	}
}

# Event for RAJ ending civil war if Pakistan hasn't been dealt with.
country_event = {
	id = mtg_britain.74
	title = mtg_britain.74.t
	desc = mtg_britain.74.d
	picture = GFX_report_event_generic_sign_treaty2

	trigger = {
		tag = RAJ
		has_war_with = PAK
		has_country_flag = RAJ_pakistani_civil_war_flag
	}
	
	#bit of code I added to make things not dumb dumb
	immediate = {
		hidden_effect = {
			remove_ideas = standing_alone
			PAK = { remove_ideas = standing_alone }
		}
	}


	fire_only_once = yes

	mean_time_to_happen = {
		days = 270
		modifier = {
			factor = 0.8
			surrender_progress > 0.2
		}
		modifier = {
			factor = 0.5
			surrender_progress > 0.5
		}
		modifier = {
			factor = 0.2
			surrender_progress > 0.8
		}
		modifier = {
			factor = 0.1
			surrender_progress > 0.9
		}
	}
	
	# Let's make peace
	option = {
		name = mtg_britain.74.a
		every_state = {
			limit = {
				is_owned_by = ROOT
				is_core_of = PAK
				is_controlled_by = PAK
			}
			transfer_state_to = PAK
		}
		if = {
			limit = {
				PAK = {
					owns_state = 443
					NOT = { owns_state = 444 }
				}
			}
			PAK = { transfer_state = 444 }
		}
		if = {
			limit = {
				PAK = {
					controls_state = 430
				}
			}
			BAN = { transfer_state = 430 }
			if = {
				limit = {
					PAK = {	
						controls_state = 432
						controls_state = 431
					}
				}
				BAN = {
					transfer_state = 431
					transfer_state = 432
				}
			}
		}
		every_country = {
			limit = {
				is_in_faction_with = RAJ
				has_war_with = PAK
			}
			white_peace = PAK
		}
		white_peace = PAK
		PAK = { country_event = { id = mtg_britain.75 } }
		# effect_tooltip = {
		# 	delete_unit_template_and_units = {
   		# 		division_template = "Hindu Militia"
		# 	}
		# }
	}

	# All or nothing - no peace!
	option = {
		name = mtg_britain.74.b
		ai_chance = {
			factor = 0
		}
		add_war_support = -0.1
	}
}
