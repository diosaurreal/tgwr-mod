			has_country_flag = EGY_BOP_ended
		}

		is_good = no
		fire_only_once = yes
		days_mission_timeout = 365

		timeout_effect = {
			log = "[GetLogRoot]: Decision timeout EGY_reform_mission"
			set_country_flag = EGY_BOP_ended
			if = {
				limit = {
					has_country_flag = EGY_Egypt_on_top
				}
				ENG={
					end_puppet = EGY
				}
				add_ideas = ENG_british_officers_independent_egypt
			}
			else_if = {
				limit = {
					has_country_flag = EGY_Egypt_lose
				}
				set_cosmetic_tag = EGY_kingdom
				ENG={
					set_autonomy = {
						target = EGY
						autonomous_state = autonomy_tgwr_informal_protectorate
					}
				}
				EGY={
					set_country_flag = protectorate
				}
			}
			else_if = {
				limit = {
					OR={
						has_country_flag = EGY_Britain_lose
						has_country_flag = EGY_Britain_on_top
					}
				}
				set_cosmetic_tag = EGY_kingdom
				ENG={
					set_autonomy = {
						target = EGY
						autonomous_state = autonomy_tgwr_integrated_protectorate
					}
				}
				EGY={
					set_country_flag = protectorate
				}
			}

		}

		complete_effect = {
			log = "[GetLogRoot]: Decision complete EGY_reform_mission"
			if = {
				limit = {
					has_country_flag = EGY_Egypt_win
				}
				every_state = {
					limit={
						ENG={
							is_owned_and_controlled_by = ROOT
							is_core_of = SUD
						}
					}
					transfer_state_to = EGY
				}
				ENG={
					end_puppet = EGY
				}
				add_ideas = ENG_british_officers_independent_egypt
				set_cosmetic_tag = EGY_kingdom
			}
			else_if = {
				limit = {
					has_country_flag = EGY_Britain_win
				}
				ENG={
					set_autonomy = {
						target = EGY
						autonomous_state = autonomy_dominion
					}
				}
				EGY={
					set_cosmetic_tag = EGY_dominion
					set_politics = {
						ruling_party = neutrality
						elections_allowed = no
					}
					add_popularity = {
						ideology = neutrality
						popularity = 0.3
					}
			    	create_country_leader = {
			    		name = "Henry McMahon"
			    		picture = "gfx/leaders/ENG/ENG_Henry_McMahon.png"
			    		ideology = colonial_government
		    			traits = {
		    				warmonger
		    			}
		    		}
				}
			}
		}

	}
