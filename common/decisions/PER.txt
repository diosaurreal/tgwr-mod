	PER_annex_BLC = {
		name = annex_FROM
		allowed = {
			original_tag = PER
		}
		target_trigger = {
			FROM = {
				tag = BLC
			}
		}
		visible = {
			is_subject = no
			is_neighbor_of = FROM
			NOT = {
				has_war_with = FROM
			}
		}
		available = {
			has_capitulated = no
			FROM = {
				OR = {
					is_in_faction = no

					is_subject_of = ROOT
					is_subject = no
				}
			}
		}
		fire_only_once = yes
		complete_effect = {

				limit = {
					has_subject = FROM
				}

				}

				add_state_core = 410

				}
			}
