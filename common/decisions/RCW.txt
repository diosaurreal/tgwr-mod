RCW_terrtorial_management = {
	RCW_core_states = {
		name = RCW_core_states
		icon = generic_army_support
		allowed = {
			original_tag = RUS
		}
		cost = 10
		state_target = yes
		target_array = global.RCW_core_states_array
		target_trigger = {
			FROM = {
				is_controlled_by = ROOT
				NOT = { is_core_of = ROOT }
				OWNER = {
					OR = {
						tag = ROOT
						has_war_with = ROOT
					}
				}
			}
		}
		target_root_trigger = {
			has_country_flag = RCW_can_core_states
		}
		on_map_mode = map_and_decisions_view
		days_remove = 10
		# available = {
		# 	FROM = {
		# 		is_controlled_by = ROOT
		# 	}
		# }
		visible = {
			has_country_flag = RCW_can_core_states
		}
		complete_effect = {

		}
		remove_effect = {
			FROM = {
				add_core_of = ROOT
				if = {
					limit = { 
						OWNER = { is_russian_tag = yes }					
					}
					if = {
						limit = {
							is_capital = no
							OWNER = {
								any_owned_state = {
									NOT = { state = FROM }
								}
							}				
						}
						add_contested_owner = OWNER
						set_state_owner_to = ROOT
					}
					else = {
						add_contested_owner = ROOT
					}
				}
			}
			every_other_country = {
				limit = {
					original_tag = ROOT
					FROM = { is_core_of = PREV }
				}
				remove_state_core = FROM
				add_state_claim = FROM
			}
		}
		ai_will_do = {
			base = 100
		}
	}
	RCW_transfer_states = {
		name = RCW_transfer_states
		icon = generic_army_support
		allowed = {
			original_tag = RUS
		}
		cost = 5
		state_target = yes
		target_array = global.RCW_non_core_states_array 
		target_trigger = {
			FROM = {
				is_controlled_by = ROOT
				NOT = { is_owned_by = ROOT }
				OWNER = {
					is_russian_tag = yes
					has_war_with = ROOT
				}
			}
		}
		target_root_trigger = {
			has_country_flag = RCW_can_core_states
		}
		on_map_mode = map_and_decisions_view
		days_remove = 5
		# available = {
		# 	FROM = {
		# 		is_controlled_by = ROOT
		# 	}
		# }
		visible = {
			has_country_flag = RCW_can_core_states
		}
		complete_effect = {

		}
		remove_effect = {
			FROM = {
				if = {
					limit = { 
						OWNER = { is_russian_tag = yes }					
					}
					if = {
						limit = { 
							OWNER = {
								any_owned_state = {
									NOT = { state = FROM }
								}
							}				
						}
						add_contested_owner = OWNER
						add_claim_by = OWNER
						set_state_owner_to = ROOT
					}
					else = {
						add_claim_by = ROOT
						add_contested_owner = ROOT
					}
				}
				if = {
					limit = { 
						has_resistance = yes
					}
					add_compliance = 20
				}
			}	
		}
		ai_will_do = {
			base = 100
		}
	}
}
