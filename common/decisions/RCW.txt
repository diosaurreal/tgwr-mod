		icon = generic_confiscation

		allowed = {
			always = no
		}
		on_map_mode = map_and_decisions_view
		highlight_states = {
			highlight_provinces = { 6338 }
		}
		available = {
			OR = {
				controls_province = 6338
				any_allied_country = {
					NOT = { has_active_mission = RCW_capture_the_gold_reserves }
					controls_province = 6338
				}

			}
			
		}

		cancel_trigger = {
			OR = {
				any_other_country = {
					has_dynamic_modifier = { modifier = RCW_gold_reserve_dynamic_modifier }

					any_allied_country = {
						owns_state = 249

					}
				}
			}
			
		}

		is_good = no

		timeout_effect = {
