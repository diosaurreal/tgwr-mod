ww1_liman_affair_category = {
	ww1_liman_affair_mission = {
		activation = {
			OR = {
				tag = TUR
				tag = GER
				has_country_flag = protest_against_liman_mission
			}
			has_global_flag = liman_mission_active
			has_global_flag = liman_crisis_alert
			NOT = {
				has_global_flag = great_war
				has_war_with_major = yes
			}
		}
		available = {
			always = no
		}
		cancel_trigger = {
			OR = {
				has_global_flag = great_war
				has_war_with_major = yes
			}
		}
		icon = mission_icon
		is_good = yes
		days_mission_timeout = 31
		timeout_effect = {
			if = {
				limit = {
					RUS = {
						has_country_flag = protest_against_liman_mission
					}
				}
				country_event = {
					id = ww1_liman_affair.10
					days = 2
				}
			}
			if = {
				limit = {
					RUS = {
						NOT = {
							has_country_flag = protest_against_liman_mission
						}
					}
				}
				country_event = {
					id = ww1_liman_affair.11
					days = 2
				}
			}
		}
	}
}
