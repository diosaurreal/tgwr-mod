defined_text = {
	name = AUS_GetParliamentRelations
	# Extremely friendly: =2 or >2
	text = {
		trigger = {
			OR = {
				check_variable = {
					AUS_parliament_relations = 2
				}
				check_variable = {
					AUS_parliament_relations > 2
				}
			}
		}
		localization_key = AUS_extremely_friendly
	}
	# Friendly: =1 or >1 up to <2
	text = {
		trigger = {
			AND = {
				OR = {
					check_variable = {
						AUS_parliament_relations = 1
					}
					check_variable = {
						AUS_parliament_relations > 1
					}
				}
				check_variable = {
					AUS_parliament_relations < 2
				}
			}
		}
		localization_key = AUS_friendly
	}
	# Unfriendly: >-0.9 and <-0.9, includes 0
	text = {
		trigger = {
			AND = {
				check_variable = {
					AUS_parliament_relations > -0.9
				}
				check_variable = {
					AUS_parliament_relations < 0.9
				}
			}
		}
		localization_key = AUS_unfriendly
	}
	# Outright hostile: =-1 or >-2 up to <-0.9
	text = {
		trigger = {
			AND = {
				OR = {
					check_variable = {
						AUS_parliament_relations = -1
					}
					check_variable = {
						AUS_parliament_relations > -2
					}
				}
				check_variable = {
					AUS_parliament_relations < -0.9
				}
			}
		}
		localization_key = AUS_outright_hostile
	}
	# Dysfunctional: =-2 or <-2
	text = {
		trigger = {
			OR = {
				check_variable = {
					AUS_parliament_relations = -2
				}
				check_variable = {
					AUS_parliament_relations < -2
				}
			}
		}
		localization_key = AUS_dysfunctional
	}
	
}
defined_text = {
    name = AUS_GetParliamentInfluence

    # Dominant: =4 or >4
    text = {
        trigger = {
            OR = {
                check_variable = { AUS_parliament_influence = 4 }
                check_variable = { AUS_parliament_influence > 4 }
            }
        }
        localization_key = AUS_Dominant
    }

    # Fairly High: =3 or >3 up to <4
    text = {
        trigger = {
            AND = {
                OR = {
                    check_variable = { AUS_parliament_influence = 3 }
                    check_variable = { AUS_parliament_influence > 3 }
                }
                check_variable = { AUS_parliament_influence < 4 }
            }
        }
        localization_key = AUS_Fairly_High
    }

    # Limited: =2 or >2 up to <3
    text = {
        trigger = {
            AND = {
                OR = {
                    check_variable = { AUS_parliament_influence = 2 }
                    check_variable = { AUS_parliament_influence > 2 }
                }
                check_variable = { AUS_parliament_influence < 3 }
            }
        }
        localization_key = AUS_Limited
    }

    # Marginal: =1 or >1 up to <2
    text = {
        trigger = {
            AND = {
                OR = {
                    check_variable = { AUS_parliament_influence = 1 }
                    check_variable = { AUS_parliament_influence > 1 }
                }
                check_variable = { AUS_parliament_influence < 2 }
            }
        }
        localization_key = AUS_Marginal
    }

    # Non-existent: =0 or >0 up to <1
    text = {
        trigger = {
            AND = {
                OR = {
                    check_variable = { AUS_parliament_influence = 0 }
                    check_variable = { AUS_parliament_influence > 0 }
                }
                check_variable = { AUS_parliament_influence < 1 }
            }
        }
        localization_key = AUS_Non_Existent
