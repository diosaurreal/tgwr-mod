on_actions = {
	on_startup = {
		effect = {
			PER = {
				PER_Persia_political = yes
				PER_init_parliament = yes
				remove_ideas = PER_the_old_persian_army_tt
				#PER_population = yes
				#PER_workshops_land = yes
				#PER_farm_land = yes
				#PER_market_workshop = yes
			}
		}
	}
	on_new_term_election = { 
		effect = {
			if = {
				limit = {
					tag = PER
				}
				PER = {
					PER_Persia_political = yes
					PER_init_parliament = yes
				}
			}
		}
	}
	on_daily = {
		effect = {
			PER_population = yes
			PER_workshops_land = yes
			PER_farm_land = yes
			PER_market_workshop = yes
			PER_Persia_ruling_party_gfx_icon = yes
			#PER_init_parliament= yes
			####
			if = {
				limit = {
					FROM = {
						original_tag = PER
						is_ai = yes
						has_country_flag = PER_gendarmerie_flag
					}
					410 = { 
						is_fully_controlled_by = FROM 
						OR = {
							is_owned_by = ENG
							is_owned_by = RAJ
						}
					}
					PER = {
						is_ai = yes
					}
					has_global_flag = persian_gendarmerie_revolt
				}
				410 = {
					transfer_state_to = FROM
				}
			}
			if = {
				limit = {
					FROM = {
						original_tag = PER
						is_ai = yes
						has_country_flag = PER_gendarmerie_flag
					}
					410 = { 
						OR = {
							is_fully_controlled_by = ENG
							is_fully_controlled_by = RAJ
						}
						is_owned_by = FROM
					}
					PER = {
						is_ai = yes
					}
					has_global_flag = persian_gendarmerie_revolt
				}
				410 = {
					transfer_state_to = ENG
				}
			}
		}
	}
	on_weekly = {
		effect = {
			Persian_Government_reforms = yes
		}
	}
	on_state_control_changed = {
		effect = {
		    if = {
				limit = {
					ROOT = { NOT = { original_tag = PER } }
					413 = { has_dynamic_modifier = { modifier = PER_khazal_crisis } }
				}
				413 = {
					remove_dynamic_modifier = { modifier = PER_khazal_crisis }
					set_state_flag = PER_khazal_crisis_no_control
				}
				PER = { remove_dynamic_modifier = { modifier = PER_khazal_crisis_modifier } }
			}
			else_if = {-
				limit = {
					ROOT = { original_tag = PER }
					413 = { has_state_flag = PER_khazal_crisis_no_control }
				}
				413 = {
					add_dynamic_modifier = { modifier = PER_khazal_crisis }
                    clr_state_flag = PER_khazal_crisis_no_control
				}
				PER = { add_dynamic_modifier = { modifier = PER_khazal_crisis_modifier } }
			}
		}
	}
}
