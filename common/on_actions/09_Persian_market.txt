on_actions = {
	on_startup = {
		effect = {
			PER = {
				remove_ideas = PER_the_old_persian_army_tt
			}
		}
	}
	on_daily = {
		effect = {
			PER_market_workshop = yes
		}
	}
	on_state_control_changed = {
		effect = {
		    if = {
				limit = {
					ROOT = { NOT = { original_tag = PER } }
					FROM.FROM = { has_dynamic_modifier = { modifier = PER_khazal_crisis } }
				}
				FROM.FROM = {
					remove_dynamic_modifier = { modifier = PER_khazal_crisis }
					set_state_flag = PER_khazal_crisis_no_control
				}
				every_country = {
					limit = { has_dynamic_modifier = { modifier = PER_khazal_crisis_modifier } }
					remove_dynamic_modifier = { modifier = PER_khazal_crisis_modifier }
				}
			}
			else_if = {-
				limit = {
					ROOT = { original_tag = PER }
					FROM.FROM = { has_state_flag = PER_khazal_crisis_no_control }
				}
				FROM.FROM = {
					add_dynamic_modifier = { modifier = PER_khazal_crisis }
                    clr_state_flag = PER_khazal_crisis_no_control
				}
				ROOT = { add_dynamic_modifier = { modifier = PER_khazal_crisis_modifier } }
			}
		}
	}
}
