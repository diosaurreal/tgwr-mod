HUN_build_army = {
	allowed = {
		original_tag = HUN
	}
	enable = {
		is_subject_of = AUS
		has_army_manpower = {
			size < 500000
		}
		NOT = {
			AND = {
				has_war_with_major = yes
				OR = {
					stockpile_ratio = {
						archetype = infantry_equipment
						ratio < 0.1
					}
					stockpile_ratio = {
						archetype = field_gun_equipment
						ratio < 0.1
					}
				}
			}
			has_global_flag = great_war_ends
			has_global_flag = crisis_in_progress
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = build_army
		id = infantry
		value = 1000
	}
	ai_strategy = {
		type = force_build_armies
		value = 1000
	}
}
