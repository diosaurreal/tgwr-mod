BUL_defend_thessaloniki = {
	allowed = {
		original_tag = BUL
	}
	enable = {
		is_great_war_ongoing = yes
		has_war = yes
		is_in_faction_with = GER
		any_neighbor_country = {
			has_war_with = FROM
			OR = {
				is_in_faction_with = ENT
				tag = ENT
			}
			NOT = {
				tag = ROM
				tag = RUS
			}
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = front_control	
		country_trigger = {
			has_war_with = BUL
			is_neighbor_of = BUL
			NOT = {
				tag = ROM
				tag = RUS
			}
		}
		ratio = 0.4
		execution_type = balanced
	}
	ai_strategy = {
		type = garrison	
		value = -200
	}
}

BUL_fight_ROM = {
	allowed = {
		original_tag = BUL
	}
	enable = {
		is_great_war_ongoing = yes
		NOT = {
			is_in_faction_with = ROM
		}
		OR = {
			has_war_with = ROM
			AND = {
				is_in_faction_with = GER
				ROM = {
					NOT = {
						has_completed_focus = ROM_turn_west
					}
					focus_progress = {
						focus = ROM_turn_west
						progress > 0
					}
				}
			}
			AND = {
				is_in_faction_with = ENT
				ROM = {
					NOT = {
						has_completed_focus = ROM_our_call_to_arms
					}
					focus_progress = {
						focus = ROM_our_call_to_arms
						progress > 0
					}
				}
			}
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = prepare_for_war
		id = ROM
		value = 1
	}
	ai_strategy = {
		type = front_control
		ratio = 0.6
		tag = ROM
		execution_type = rush
	}
	ai_strategy = {
		type = garrison	
		value = -1000
	}
	ai_strategy = {
		type = force_defend_ally_borders
		target = eastern_balkans
		value = 100
	}
}
