BUL_defend_thessaloniki = {
	allowed = {
		original_tag = BUL
	}
	enable = {
		is_great_war_ongoing = yes
		has_war = yes
		is_in_faction_with = GER
		any_neighbor_country = {
			is_in_faction_with = ENT
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = front_unit_request	
		country_trigger = {
			OR = {
				tag = THE
				tag = GRE
			}
		}
		state_trigger = {
			OR = {
				is_core_of = GRE
				is_core_of = BUL
			}
		}
		value = 300
	}
	ai_strategy = {
		type = front_control	
		country_trigger = {
			OR = {
				tag = THE
				tag = GRE
			}
		}
		state_trigger = {
			OR = {
				is_core_of = GRE
				is_core_of = BUL
			}
		}
		ratio = 0.8
		execution_type = balanced
	}
	ai_strategy = {
		type = garrison	
		value = -200
	}
}
