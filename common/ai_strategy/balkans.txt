SER_GRE_dont_put_units_in_bulgaria = {
	allowed = {
		OR = {
			original_tag = SER
			original_tag = GRE
			original_tag = MNT
			original_tag = CRE
			original_tag = SMS
			original_tag = IKR
		}
	}
	enable = {
		is_in_faction_with = BUL
		is_great_war_ongoing = no
	}
	abort_when_not_enabled = yes

	abort = {
		OR = {
			has_war_with = AUS
			BUL = {
				OR = {
					surrender_progress > 0.05
					casualties > 25000
				}
			}
			NOT = {has_border_war_with = TUR}
		}
	}
	
	ai_strategy = {
		type = dont_defend_ally_borders
		id = BUL
		value = 200
	}
}
