RAJ_prepare_aden_front = {
	allowed = {
		original_tag = RAJ
	}
	enable = {
		controls_state = 659
		TUR = {
			OR = {
				has_active_mission = TUR_black_sea_raid_mission
				AND = {
					focus_progress = {
						focus = TUR_german_ottoman_alliance
						progress > 0
					}
					ROOT = {
						has_war_with = GER
					}
				}
				AND = {
					focus_progress = {
						focus = TUR_our_traditional_allies
						progress > 0
					}
					ROOT = {
						has_war_with = ENT
					}
				}
				any_allied_country = {
					OR = {
						has_wargoal_against = TUR
						has_war_with = TUR
						is_justifying_wargoal_against = TUR
						TUR = {
							OR = {
								is_justifying_wargoal_against = PREV
								has_wargoal_against = PREV
							}
						}
					}
				}
				TUR = {
					OR = {
						is_justifying_wargoal_against = ROOT
						has_wargoal_against = ROOT
					}
				}
			}
		}
		NOT = {
			has_war_with = TUR
			is_in_faction_with = TUR
			has_global_flag = great_war_ends
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = prepare_for_war
		id = TUR 
		value = 1
	}
	ai_strategy = { # shitty hoi AI think front with NAJ is the same as front with TUR
		type = prepare_for_war
		id = NAJ
		value = 1
	}
	ai_strategy = {
		type = area_priority
		id = europe
		value = -1000
	}
	ai_strategy = {
		type = garrison
		value = -100
	}
	ai_strategy = {
		type = garrison_reinforcement_priority
		value = -100
	}
	ai_strategy = {
		type = front_unit_request
		strategic_region = 196
		value = 350
	}
	ai_strategy = {
		type = front_unit_request
		tag = GER
		value = -1000
	}
	ai_strategy = {
		type = dont_defend_ally_borders
