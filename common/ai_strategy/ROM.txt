	}
	enable = {	
		NOT = {
			has_war_with = RUS
			has_war_with = BUL

		}	
		OR = {
			date > 1914.1.1
			threat > 0.5

		type = antagonize
		id = "AUS"
		value = 100
	}

		type = alliance
		id = AUS
		value = -200
	}
}
