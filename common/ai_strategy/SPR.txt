SPR_trade_with_Entente = {
	allowed = {
		original_tag = SPR
	}
	enable = {
		NOT = {
			has_government = communism
			has_war_with = FRA
			has_war_with = ENG
			is_in_faction_with = GER
		}	
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = equipment_market_trade_desire
		id = ENG
		value = 100
	}
	ai_strategy = {
		type = equipment_market_trade_desire
		id = FRA
		value = 100
	}
}
