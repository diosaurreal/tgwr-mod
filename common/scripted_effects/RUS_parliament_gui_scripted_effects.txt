# The following scripted effect sets-up the parliament at game start.
RUS_parliament_gui_initial = {
	# Put simply an array stores a list of things, used here to store the list of parties and list of seats.
	# parliament_seat_array stores the list of seat numbers for each party
	add_to_array = { RUS_parliament_seat_array = socialists }
	add_to_array = { RUS_parliament_seat_array = center_left }
	add_to_array = { RUS_parliament_seat_array = center_right }
	add_to_array = { RUS_parliament_seat_array = far_right }


	add_to_array = { RUS_parliament_member_array = 1 } # socialists
	add_to_array = { RUS_parliament_member_array = 2 } # center_left
	add_to_array = { RUS_parliament_member_array = 3 } # center_right
	add_to_array = { RUS_parliament_member_array = 4 } # far_right


	# Set up the positions of the dots for the semicircle parliament
	RUS_parliament_gui_positions = yes

	# Modify what shows up in the gui
	RUS_reload_parliament_gui = yes
}

#Example how to use RUS_modify_state_council
#  set_temp_variable = { RUS_modify_state_council_var = -2 }
#  RUS_modify_state_council = yes
#Changes dominant faction var in State Council by -2
RUS_modify_state_council = {
	if = {
		limit = {
			OR = {
				check_variable = {
					RUS_modify_state_council_var > 5
				}
				check_variable = {
					RUS_modify_state_council_var < -5
				}
			}
		}
		custom_effect_tooltip = RUS_modify_state_council_big_tt
	}
	else = {
		custom_effect_tooltip = RUS_modify_state_council_tt
	}
	add_to_variable = { state_council_dominant_faction = RUS_modify_state_council_var }
	clamp_variable = {
		var = state_council_dominant_faction
		min = 0
		max = 100
	}
}
RUS_state_council_election_effect = {
	hidden_effect = {
		set_temp_variable = { RUS_democratic_popularity = party_popularity_100@democratic }
		set_temp_variable = { RUS_neutrality_popularity = party_popularity_100@neutrality }
		set_temp_variable = { RUS_fascism_popularity = party_popularity_100@fascism }
		set_temp_variable = { RUS_communism_popularity = party_popularity_100@communism }
		divide_temp_variable = { RUS_communism_popularity = 4 }
		divide_temp_variable = { RUS_democratic_popularity = 4 }
		divide_temp_variable = { RUS_neutrality_popularity = 4 }
		divide_temp_variable = { RUS_fascism_popularity = 4 }
		if = {
			limit = {
				communism_plurality = yes
			}
			set_temp_variable = { RUS_state_council_election_result = 12.5 }
			add_to_temp_variable = { RUS_state_council_election_result = RUS_democratic_popularity }
			add_to_temp_variable = { RUS_state_council_election_result = RUS_neutrality_popularity }
			add_to_temp_variable = { RUS_state_council_election_result = RUS_fascism_popularity }
			round_temp_variable = RUS_state_council_election_result
		}
		else_if = {
			limit = {
				democratic_plurality = yes
			}
			set_temp_variable = { RUS_state_council_election_result = 37.5 }
			add_to_temp_variable = { RUS_state_council_election_result = RUS_neutrality_popularity }
			add_to_temp_variable = { RUS_state_council_election_result = RUS_fascism_popularity }
			subtract_from_temp_variable = { RUS_state_council_election_result = RUS_communism_popularity }
			round_temp_variable = RUS_state_council_election_result
		}
		else_if = {
			limit = {
				neutrality_plurality = yes
			}
			set_temp_variable = { RUS_state_council_election_result = 62.5 }
			subtract_from_temp_variable = { RUS_state_council_election_result = RUS_communism_popularity }
			subtract_from_temp_variable = { RUS_state_council_election_result = RUS_democratic_popularity }
			add_to_temp_variable = { RUS_state_council_election_result = RUS_fascism_popularity  }
			round_temp_variable = RUS_state_council_election_result
		}
		else_if = {
			limit = {
				fascism_plurality = yes
			}
			set_temp_variable = { RUS_state_council_election_result = 87.5 }
			subtract_from_temp_variable = { RUS_state_council_election_result = RUS_communism_popularity }
			subtract_from_temp_variable = { RUS_state_council_election_result = RUS_democratic_popularity }
			subtract_from_temp_variable = { RUS_state_council_election_result = RUS_neutrality_popularity }
		}
	}
	round_temp_variable = RUS_state_council_election_result
	set_variable = { state_council_dominant_faction = RUS_state_council_election_result }
	custom_effect_tooltip = RUS_state_council_election_tt
}
# Duma election scripted effect
RUS_election_scripted_effect = {
	custom_effect_tooltip = RUS_election_tt
	hidden_effect = {
		set_temp_variable = { communism_temp = party_popularity@communism }
		multiply_temp_variable = { communism_temp = 442 }

		set_temp_variable = { democratic_temp = party_popularity@democratic }
		multiply_temp_variable = { democratic_temp = 442 }


		set_temp_variable = { neutrality_temp = party_popularity@neutrality }
		multiply_temp_variable = { neutrality_temp = 442 }


		set_temp_variable = { fascism_temp = party_popularity@fascism }
		multiply_temp_variable = { fascism_temp = 442 }

		add_to_temp_variable = {
			neutrality_temp = 1 #because rounding var cause 1 seat to dissapear, so this way it come back
		}

		set_variable = { var = far_right value = fascism_temp }
		set_variable = { var = center_right value = neutrality_temp }
		set_variable = { var = center_left value = democratic_temp }
		set_variable = { var = socialists value = communism_temp }
		round_variable = far_right
		round_variable = center_right
		round_variable = center_left
		round_variable = socialists

		RUS_reload_loyalties = yes

		set_variable = { RUS_parliament_seat_array^0 = socialists }
		set_variable = { RUS_parliament_seat_array^1 = center_left }
		set_variable = { RUS_parliament_seat_array^2 = center_right }
		set_variable = { RUS_parliament_seat_array^3 = far_right }
		RUS_reload_parliament_gui = yes
	}
}

RUS_modify_ruling_loyalty = {
	if = {
		limit = { has_government = communism }
		set_temp_variable = { RUS_modify_loyalty_socialists_var = RUS_modify_ruling_loyalty_var }
		RUS_modify_loyalty_socialists = yes
	}
	if = {
		limit = { has_government = democratic }
		set_temp_variable = { RUS_modify_loyalty_center_left_var = RUS_modify_ruling_loyalty_var }
		RUS_modify_loyalty_center_left = yes
	}

		limit = { has_government = neutrality }
		set_temp_variable = { RUS_modify_loyalty_center_right_var = RUS_modify_ruling_loyalty_var }
		RUS_modify_loyalty_center_right = yes
	}
	if = {
		limit = { has_government = fascism }
		set_temp_variable = { RUS_modify_loyalty_far_right_var = RUS_modify_ruling_loyalty_var }
		RUS_modify_loyalty_far_right = yes
	}
	RUS_reload_loyalties = yes
}
#Sway some MPs to another Faction
RUS_recalculate_sway = {
	clamp_temp_variable = {
		var = sway_value_tt
		min = 0
		max = RUS_parliament_seat_array^parliament_loser
	}
	set_variable = {
		var = sway_value
		value = sway_value_tt
	}
	RUS_remove_parliament_members_from_temp_vars = yes
	RUS_add_parliament_members_from_temp_vars = yes
	RUS_reload_loyalties = yes
	custom_effect_tooltip = rus_parliament_sway_tooltip
}

RUS_remove_parliament_members_from_temp_vars = {
	if = {
		limit = {
			check_variable = { var = parliament_loser value = 1 compare = less_than }
		}
		clamp_variable = {
			var = sway_value
			min = 0
			max = socialists
		}
		subtract_from_variable = {
			var = socialists
			value = sway_value
		}
		
	}
	if = {
		limit = {
			check_variable = { var = parliament_loser value = 1 compare = equals }
		}
		clamp_variable = {
			var = sway_value
			min = 0
			max = center_left
		}
		subtract_from_variable = {
			var = center_left
			value = sway_value
		}
	}
	if = {
		limit = {
			check_variable = { var = parliament_loser value = 2 compare = equals }
		}
		clamp_variable = {
			var = sway_value
			min = 0
			max = center_right
		}
		subtract_from_variable = {
			var = center_right
			value = sway_value
		}
	}
	if = {
		limit = {
			check_variable = { var = parliament_loser value = 3 compare = greater_than_or_equals }
		}
		clamp_variable = {
			var = sway_value
			min = 0
			max = far_right
		}
		subtract_from_variable = {
			var = far_right
			value = sway_value
		}
	}
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_socialists = socialists }
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_center_left = center_left }
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_center_right = center_right }
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_far_right = far_right }
	RUS_reload_parliament_gui = yes
}

RUS_add_parliament_members_from_temp_vars = {
	if = {
		limit = {
			check_variable = { var = parliament_winner value = 1 compare = less_than }
		}
		add_to_variable = {
			var = socialists
			value = sway_value
		}
	}
	if = {
		limit = {
			check_variable = { var = parliament_winner value = 1 compare = equals }
		}
		add_to_variable = {
			var = center_left
			value = sway_value
		}
	}
	if = {
		limit = {
			check_variable = { var = parliament_winner value = 2 compare = equals }
		}
		add_to_variable = {
			var = center_right
			value = sway_value
		}
	}
	if = {
		limit = {
			check_variable = { var = parliament_winner value = 3 compare = greater_than_or_equals }
		}
		add_to_variable = {
			var = far_right
			value = sway_value
		}
	}
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_socialists = socialists }
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_center_left = center_left }
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_center_right = center_right }
	set_variable = { RUS_parliament_seat_array^RUS_parliament_seat_far_right = far_right }
	RUS_reload_parliament_gui = yes
}

	set_variable = { var = far_right_loyalty_litteral_seats  value = far_right }
	set_variable = { var = center_right_loyalty_litteral_seats value = center_right }
	set_variable = { var = center_left_loyalty_litteral_seats  value = center_left }
	set_variable = { var = socialists_loyalty_litteral_seats value = socialists }

	multiply_variable = {
		var = far_right_loyalty_litteral_seats
		value = far_right_loyalty
	}
	divide_variable = {
		var = far_right_loyalty_litteral_seats
		value = 100
	}

	multiply_variable = {
		var = center_right_loyalty_litteral_seats
		value = center_right_loyalty
	}
	divide_variable = {
		var = center_right_loyalty_litteral_seats
		value = 100
	}

	multiply_variable = {
		var = center_left_loyalty_litteral_seats
		value = center_left_loyalty
	}
	divide_variable = {
		var = center_left_loyalty_litteral_seats
		value = 100
	}

	multiply_variable = {
		var = socialists_loyalty_litteral_seats
		value = socialists_loyalty
	}
	divide_variable = {
		var = socialists_loyalty_litteral_seats
		value = 100
	}

	set_variable = { var = total_loyalist_count value = 0 }
	add_to_variable = {
		var = total_loyalist_count
		value = far_right_loyalty_litteral_seats
	}
	add_to_variable = {
		var = total_loyalist_count

	}
	add_to_variable = {
		var = total_loyalist_count
		value = center_left_loyalty_litteral_seats
	}
	add_to_variable = {
		var = total_loyalist_count
		value = socialists_loyalty_litteral_seats
	}
}

# The following scripted effect does the main job of generating the gui, by adding the seats for each party in a grid.
# This should be called every time you need to update the gui, when you add/subtract seats for example.
# Read the comments if you're interested in how it works
RUS_reload_parliament_gui = {
	# Clears what's currently on the gui, if there's anything there.
	clear_array = RUS_parliament_main_array

	# The array temp_seat is duplicated from the array parliament_seat_array.
	for_each_loop = {
		array = RUS_parliament_seat_array

		add_to_array = { temp_seat = RUS_parliament_seat_array^i }
	}

	# It's a suprise tool for later
	set_temp_variable = { temp_index_1 = 0 }
	set_temp_variable = { temp_index_2 = 1 }

	# Set temp_size as the highest index in the array, highest index is the highest value x in temp_array^x.
	set_temp_variable = { temp_size = temp_seat^num }
	subtract_from_temp_variable = { temp_size = 1 }

	# What this does is loop through the temp_seat array, adding the current value to the next value on the list.
	# temp_seat^0 is the first value on the list, temp_seat^1 is the next and so forth.
	# the loop ends when temp_index_1 is equal to the size of the array - 1 (basically the max index).
	while_loop_effect = {
		limit = {
			check_variable = { temp_index_1 < temp_size }
		}
		# The suprise tool starts out as temp_seat^0 and temp_seat^1 and then is incremented every loop , adding the current value to the next value on the list.
		add_to_variable = { temp_seat^temp_index_2 = temp_seat^temp_index_1 }
		add_to_temp_variable = { temp_index_1 = 1 }
		add_to_temp_variable = { temp_index_2 = 1 }
	}

	# resize_array fills the parliament_member_array with the id of the parliament member x amount of times where x is the number of seats the member has.
	# so the array will look like this basically:
	# 1,1,1,1,1,1,1,...(repeated 90 times),2,2,2,2,2,2,2(repeated 35 times),3,3,3.. etc.
	for_loop_effect = {
		start = 0
		end = temp_seat^num

		resize_array = { array = RUS_parliament_main_array value = RUS_parliament_member_array^v size = temp_seat^v }
	}

	# Clear the temp_seat array used previously since it's not needed anymore.
	clear_array = temp_seat
}
