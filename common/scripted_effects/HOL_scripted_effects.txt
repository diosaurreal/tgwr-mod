HOL_increase_arp_pop = {
	if = {
		limit = {
			check_variable = {
				HOL_variable_ag_popularity > HOL_variable_lu_popularity				
			}
			check_variable = {
				HOL_variable_ag_popularity > HOL_variable_sdap_popularity				
			}
		}
		add_to_variable = {
			HOL_variable_ag_popularity = -3
		}
	}
	if = {
		limit = {
			check_variable = {
				HOL_variable_lu_popularity > HOL_variable_ag_popularity				
			}
			check_variable = {
				HOL_variable_lu_popularity > HOL_variable_sdap_popularity				
			}
		}
		add_to_variable = {
			HOL_variable_lu_popularity = -3
		}
	}
	if = {
		limit = {
			check_variable = {
				HOL_variable_sdap_popularity > HOL_variable_ag_popularity				
			}
			check_variable = {
				HOL_variable_sdap_popularity > HOL_variable_lu_popularity				
			}
		}
		add_to_variable = {
			HOL_variable_sdap_popularity = -3
		}
	}
	add_to_variable = {
		HOL_variable_arp_popularity = 3
	}
	clamp_variable = {
    	var = HOL_variable_arp_popularity
    	min = 0
		max = 100
	}
	clamp_variable = {
    	var = HOL_variable_sdap_popularity
    	min = 0
		max = 100
	}
	clamp_variable = {
    	var = HOL_variable_lu_popularity
    	min = 0
		max = 100
	}
	clamp_variable = {
    	var = HOL_variable_ag_popularity
    	min = 0
		max = 100
	}
}
