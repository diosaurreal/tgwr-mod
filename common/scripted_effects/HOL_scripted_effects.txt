HOL_increase_arp_pop2 = {
	find_highest_in_array = {
    	array = HOL_array_parties_noarp
    	value = HOL_temp_var
	}
	if = {
		limit = {
			check_var = {
				var = HOL_variable_sdap_popularity
				value = HOL_temp_var
				compare = equals_to
 			}
		}
		subtract_from_variable = {
			var = HOL_variable_sdap_popularity
			value = 3
 		}
	}
	if = {
		limit = {
			check_var = {
				var = HOL_variable_ag_popularity
				value = HOL_temp_var
				compare = equals_to
 			}
		}
		subtract_from_variable = {
			var = HOL_variable_ag_popularity
			value = 3
 		}
	}
	if = {
		limit = {
			check_var = {
				var = HOL_variable_lu_popularity
				value = HOL_temp_var
				compare = equals_to
 			}
		}
		subtract_from_variable = {
			var = HOL_variable_lu_popularity
			value = 3
 		}
	}
	add_to_variable = {
		HOL_variable_arp_popularity = 3
	}
	clamp_variable = {
    	var = HOL_variable_arp_popularity
    	min = 0
		max = 100
	}
	clamp_variable = {
    	var = HOL_variable_sdap_popularity
    	min = 0
		max = 100
	}
	clamp_variable = {
    	var = HOL_variable_lu_popularity
    	min = 0
		max = 100
	}
	clamp_variable = {
    	var = HOL_variable_ag_popularity
    	min = 0
		max = 100
	}
}
