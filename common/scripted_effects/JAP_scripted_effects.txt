JAP_setup_ship_targets = {
	set_variable = {
		jap_target_battleship = num_ships_with_type@battleship
	}
	set_variable = {
		jap_target_battlecruiser = num_ships_with_type@battle_cruiser
	}
	set_variable = {
		jap_target_protected_cruiser =  num_ships_with_type@protected_cruiser
	}
	set_variable = {
		jap_target_armored_cruiser =  num_ships_with_type@armored_cruiser
	}
	set_variable = {
		jap_target_unprotected_cruiser =  num_ships_with_type@unprotected_cruiser
	}
	set_variable = {
		jap_target_light_cruiser =  num_ships_with_type@light_cruiser
	}
	set_variable = {
		jap_target_destroyer = num_ships_with_type@destroyer
	}
}

	JAP_setup_ship_targets = yes
	add_to_variable = {
		jap_target_battleship = 1
	}
	add_to_variable = {
		jap_target_battlecruiser = 1
	}
	add_to_variable = {
		jap_target_protected_cruiser = 1
	}
	add_to_variable = {
		jap_target_destroyer = 1
	}	
	activate_mission = JAP_finish_old_naval_projects
}

JAP_finish_uncompleted_plan_effect = {
	hidden_effect = {
		set_country_flag = failed_naval_program
	}
	if = {
		limit = { 
			has_variable = jap_target_battleship
		}
		set_temp_variable = {
			difference_bb = jap_target_battleship
		}
		subtract_from_temp_variable = {
			difference_bb = num_ships_with_type@battleship
		}
		set_variable = {
			jap_target_battleship = num_ships_with_type@battleship
		}
		add_to_variable = {
			jap_target_battleship = difference_bb
		}
	}
	if = {
		limit = { 
			has_variable = jap_target_battlecruiser
		}
		set_temp_variable = {
			difference_bc = jap_target_battlecruiser
		}
		subtract_from_temp_variable = {
			difference_bc = num_ships_with_type@battle_cruiser
		}
		set_variable = {
			jap_target_battlecruiser = num_ships_with_type@battle_cruiser
		}
		add_to_variable = {
			jap_target_battlecruiser = difference_bc
		}
	}
	if = {
		limit = { 
			has_variable = jap_target_protected_cruiser
		}
		set_temp_variable = {
			difference_pc = jap_target_protected_cruiser
		}
		subtract_from_temp_variable = {
			difference_pc = num_ships_with_type@protected_cruiser
		}
		set_variable = {
			jap_target_protected_cruiser = num_ships_with_type@protected_cruiser
		}
		add_to_variable = {
			jap_target_protected_cruiser = difference_pc
		}
	}
	if = {
		limit = { 
			has_variable = jap_target_destroyer
		}
		set_temp_variable = {
			difference_dd = jap_target_destroyer
		}
		subtract_from_temp_variable = {
			difference_dd = num_ships_with_type@destroyer
		}
		set_variable = {
			jap_target_destroyer = num_ships_with_type@destroyer
		}
		add_to_variable = {
			jap_target_destroyer = difference_dd
		}
	}
	activate_mission = JAP_finish_uncompleted_plan
}
