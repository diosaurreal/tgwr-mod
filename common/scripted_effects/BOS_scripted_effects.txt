BOS_increase_serb_opinion = {
	if = {
		limit = {
			check_variable = {
				BOS_serb_opinion < 5
			}
		}
		add_to_variable = {
			BOS_serb_opinion = 1
		}
		custom_effect_tooltip = BOS_increase_serb_tt
		else = {
			add_stability = 0.02
			add_political_power = 25
		}
	}
	BOS_clamp_opinion_vars = yes
}
BOS_decrease_serb_opinion = {
