BOS_increase_serb_opinion = {
	if = {
		limit = {
			check_variable = {
				BOS_serb_opinion < 5
			}
		}
		add_to_variable = {
			BOS_serb_opinion = 1
		}
		custom_effect_tooltip = BOS_increase_serb_tt
		else = {
			add_stability = 0.02
			add_political_power = 25
		}
	}
	BOS_clamp_opinion_vars = yes
}

BOS_first_diet_uncompleted = {
	set_variable = {
		BOS_uncompleted_focus = 0
	}
	if = {
		limit = {
			NOT = {
				has_completed_focus = BOS_road_construction_bill
			}
		}
		custom_effect_tooltip = modify_BOS_TUR_tt
		add_to_variable = {
			BOS_TUR_construction_speed = -0.1
			tooltip = production_speed_buildings_factor_tt
		}
		add_to_variable = {
			BOS_uncompleted_focus = 1
		}
	}
	if = {
		limit = {
			NOT = {
				has_completed_focus = BOS_compulsory_education_law
			}
		}
		set_country_flag = not_completed_compulsory_education
		activate_mission = BOS_teachers_demand_better_pay
		unlock_decision_tooltip = BOS_compulsory_education_dec
		add_to_variable = {
			BOS_uncompleted_focus = 1
		}
	}
	if = {
		limit = {
			NOT = {
				has_completed_focus = BOS_form_the_control_committee
			}
		}
		add_political_power = -15
		add_to_variable = {
			BOS_uncompleted_focus = 1
		}
	}
	if = {
		limit = {
			NOT = {
				has_completed_focus = BOS_purge_german_in_communication
			}
		}
		add_political_power = -15
		add_to_variable = {
			BOS_uncompleted_focus = 1
		}
	}
	if = {
		limit = {
			NOT = {
				has_completed_focus = BOS_remove_hungarian_lang_from_railways
			}
		}
		add_political_power = -15
		add_to_variable = {
			BOS_uncompleted_focus = 1
		}
	}
	if = {
		limit = {
			NOT = {
				has_completed_focus = BOS_postal_savings_bank
			}
		}
		add_political_power = -15
		add_to_variable = {
			BOS_uncompleted_focus = 1
		}
	}
