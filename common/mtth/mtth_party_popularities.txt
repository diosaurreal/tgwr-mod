neutrality_party_popularity_var = {
	base = 0
	modifier = {
		neutrality > 0

		set_temp_variable = { neutrality_party_popularity_var_temp = 0 }
		if = {
			limit = { has_completed_focus = TUR_unity_of_the_party_and_the_committee }
			set_temp_variable = { neutrality_party_popularity_var_TUR_unity_of_the_party_and_the_committee = party_popularity@neutrality }
			multiply_temp_variable = { neutrality_party_popularity_var_TUR_unity_of_the_party_and_the_committee = 0.05 }
			add_to_temp_variable = { neutrality_party_popularity_var_temp = neutrality_party_popularity_var_TUR_unity_of_the_party_and_the_committee }
		}
		if = {
			limit = { has_idea = TUR_government_of_national_unity_idea_yes }
			set_temp_variable = { neutrality_party_popularity_var_TUR_government_of_national_unity = party_popularity@neutrality }
			multiply_temp_variable = { neutrality_party_popularity_var_TUR_government_of_national_unity = 0.1 }
			add_to_temp_variable = { neutrality_party_popularity_var_temp = neutrality_party_popularity_var_TUR_government_of_national_unity }
		}
		add = neutrality_party_popularity_var_temp
	}
}
