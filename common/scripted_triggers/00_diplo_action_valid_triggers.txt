# scope is target country

is_diplomatic_action_valid_stage_coup = {
	always = no
}

	always = no
}
	
is_diplomatic_action_valid_declare_war = {
	NOT = { 
		is_subject_of = FROM.overlord 
		AND = {
			is_subject = yes
			OVERLORD = {
				is_subject_of = FROM
			}
		}
		AND = {
			FROM = {
				is_subject = yes
				OVERLORD = {
					is_subject_of = ROOT
				}
			}
		}	
	}
}

is_diplomatic_action_valid_assume_faction_leadership = {
	has_global_flag = interbellum
	OR = {
		has_country_flag = interbellum
		NOT = {
			has_country_flag = great_war_loser
			has_country_flag = great_war_winner
		}
	}
}

is_diplomatic_action_valid_join_allies = {
	if = {
		limit = {
			OR = {
				has_autonomy_state = autonomy_tgwr_protectorate
				has_autonomy_state = autonomy_tgwr_informal_protectorate
				has_autonomy_state = autonomy_tgwr_ottoman_emirate
				has_autonomy_state = autonomy_tgwr_concession
			}			
		}
		always = no
	}
	if = {
		limit = {
			FROM = {
				OR = {
					has_autonomy_state = autonomy_tgwr_protectorate
					has_autonomy_state = autonomy_tgwr_informal_protectorate
					has_autonomy_state = autonomy_tgwr_ottoman_emirate
					has_autonomy_state = autonomy_tgwr_concession
				}
			}						
		}
		FROM = {
			has_defensive_war = yes
		}
	}
	if = {
		limit = {
			is_in_faction_with = ITA 
			FROM = {
				tag = ITA
			}
			has_global_flag = italo_turkish_war
			NOT = {
				has_global_flag = italo_turkish_war_ends 
				has_global_flag = great_war
			}
		}
		NOT = {
			has_global_flag = italo_turkish_war_ends 
			ITA = {
				any_enemy_country = {
					NOT = {
						original_tag = TUR 
						is_subject_of = TUR
					}
					is_major = yes
				}
			}
		}	
	}
	if = {
		limit = {
			is_in_faction_with = FRA 
			NOT = {
				is_subject_of = FRA
				tag = FRA
			}
			FROM = {
				has_war_with = CHA
				OR = {
					is_subject_of = FRA
					tag = FRA
				}
			}
		}
		FROM = {
			any_enemy_country = {
				NOT = {
					original_tag = CHA 
					original_tag = DAR
					original_tag = LBA
				}
			}
		}		
	}
	if = {
		limit = {
			FROM = {
				original_tag = TUR
				has_country_flag = TUR_kostantiniye_government_tag
				has_country_flag = great_war_loser
			}
			has_global_flag = great_war_ends
			OR = {
				is_in_faction_with = FROM 
				has_subject = FROM
			}
		}
		FROM = {
			any_enemy_country = {
				NOT = {
					original_tag = TUR 
					original_tag = KUR
				}
				is_major = yes
			}
		}		
	}
}

		any_enemy_country = {

					is_in_faction_with = ROOT 
					has_subject = ROOT
				}		
			}
		}
		ROOT = {

				NOT = {

				is_major = yes
			}
		}

is_diplomatic_action_valid_join_faction = {
	if = {
		limit = {
			original_tag = TRK
			is_faction_leader = yes
		}	
		FROM = { 
			OR = {
				original_tag = BUK
				original_tag = KHI
				original_tag = KAZ
				original_tag = AFG
				original_tag = PER
				original_tag = KUM
				original_tag = SIK
				original_tag = ETS
			} 	
		}
	}
	if = {
		limit = {
			original_tag = ITA
			has_country_flag = ITA_monarchist_italy_flag
            is_faction_leader = yes
		}	
		FROM = { 
			is_italian_nation = yes	
		}
	}
	if = {
		limit = {
			tag = BLL
		}	
		FROM = {
			OR = {
				is_balkan_nation = yes
				is_greek_nation = yes
			}
		}
	}
	if = {
		limit = {
			original_tag = ITA
			has_country_flag = ITA_monarchist_italy_flag
            is_faction_leader = yes
		}	
		FROM = { 
			is_italian_nation = yes	
		}
	}
	if = {
		limit = {
			original_tag = BRA
			is_faction_leader = yes
		}	
		FROM = {
			is_latin_american_country = yes
		}
	}
	if = {
		limit = {
			tag = NPA
			is_faction_leader = yes
		}	
		FROM = {
			is_chinese_warlord_no_tt = yes
		}
	}
	if = {
		limit = {
			tag = CPG
			is_faction_leader = yes
		}	
		FROM = {
			is_chinese_warlord_no_tt = yes
		}
	}
	if = {
		limit = {
			OR = {
				tag = ZHC
				tag = ANC
				tag = FNG
			}
			is_faction_leader = yes
			NOT = {
				has_country_flag = reunited_china
			}
		}	
		FROM = {
			is_chinese_warlord_no_tt = yes
		}
	}
