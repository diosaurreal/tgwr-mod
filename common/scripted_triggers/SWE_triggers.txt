SWE_can_get_neutrality_events = {
	has_global_flag = great_war
	NOT = { has_global_flag = great_war_ends }
	NOT = { has_country_flag = end_of_neutrality_opposition }
	NOT = { has_country_flag = pause_between_neutrality_events }
}

SWE_agrarian_party_split = {
	date > 1913.1.1
	OR = {
		has_country_flag = SWE_suffrage_flag
		NOT = {
			has_idea = SWE_unofficial_parliamentalism
			has_idea = SWE_restricted_parliamentalism
		}
	}
}
