RUS_liberals_satisfied = {
	custom_trigger_tooltip = {
		tooltip = RUS_liberals_satisfied_tt
		if = {
			limit = {
				has_country_flag = RUS_coalition_with_liberals
			}
			count_triggers = {
				amount = 3
				set_temp_variable = {
					RUS_agrarian_reform_progress_temp_var = 20
				}
				check_variable = {
					var = RUS_agrariran_reform_progress_var
					value = RUS_agrarian_reform_progress_temp_var
					compare = greater_than_or_equals
					tooltip = has_agrarian_reform_progress_tt
				}
				set_temp_variable = {
					RUS_education_reform_progress_temp_var = 20
				}
				check_variable = {
					var = RUS_education_reform_progress_var
					value = RUS_education_reform_progress_temp_var
					compare = greater_than_or_equals
					tooltip = has_education_reform_progress_tt
				}
				set_temp_variable = {
					RUS_labour_reform_progress_temp_var = 40
				}
				check_variable = {
					var = RUS_labour_reform_progress_var
					value = RUS_labour_reform_progress_temp_var
					compare = greater_than_or_equals
					tooltip = has_labour_reform_progress_tt
				}
				set_temp_variable = {
					RUS_zemstvo_reform_progress_temp_var = 60
				}
				check_variable = {
					var = RUS_zemstvo_reform_progress_var
					value = RUS_zemstvo_reform_progress_temp_var
					compare = greater_than_or_equals
					tooltip = has_zemstvo_reform_progress_tt
				}
			}
			NOT = {
				has_country_flag = RUS_rigged_the_election
				has_country_flag = RUS_centrists_left_the_coalition
			}
		}
		else = {
			always = no
		}
	}
}
