ALB_character_trigger = {
	if = {
		limit = { ALB_is_kosovar_character = yes }
		OR = {
			OR = {
				has_cosmetic_tag = ALB_boletini_flag
				AND = {
					NOT = { has_cosmetic_tag = ALB_boletini_flag }
					capital_scope = { ALB_is_kosovar_state = yes }
				}
			}
			AND = {
				OR = {
					has_cosmetic_tag = ALB_kimza_flag
					AND = {
						NOT = { has_cosmetic_tag = ALB_kimza_flag }
						capital_scope = { ALB_is_northern_state = yes }
					}
				}
				has_country_flag = ALB_can_organize_northern_uprisings
			}
			has_global_flag = ALB_unified_national_movement
		}
	}
	else_if = {
		limit = { ALB_is_northern_character = yes }
		OR = {
			OR = {
				has_cosmetic_tag = ALB_kimza_flag
				AND = {
					NOT = { has_cosmetic_tag = ALB_kimza_flag }
					capital_scope = { ALB_is_northern_state = yes }
				}
			}
			AND = {
				OR = {
					has_cosmetic_tag = ALB_boletini_flag
					AND = {
						NOT = { has_cosmetic_tag = ALB_boletini_flag }
						capital_scope = { ALB_is_kosovar_state = yes }
					}
				}
				has_country_flag = ALB_can_organize_northern_uprisings
			}
			has_global_flag = ALB_unified_national_movement
		}
	}
	else_if = {
		limit = { ALB_is_central_character = yes }
		OR = {
			OR = {
				has_cosmetic_tag = ALB_toptani_flag
				AND = {
					NOT = { has_cosmetic_tag = ALB_toptani_flag }
					capital_scope = { ALB_is_central_state = yes }
				}
			}
			has_global_flag = ALB_unified_national_movement
		}
	}
	else_if = {
		limit = { ALB_is_southern_character = yes }
		OR = {
			OR = {
				has_cosmetic_tag = ALB_vlora_flag
				AND = {
					NOT = { has_cosmetic_tag = ALB_vlora_flag }
					capital_scope = { ALB_is_southern_state = yes }
				}
			}
			has_global_flag = ALB_unified_national_movement
		}
	}
}

ALB_albanian_assembly_trigger = {
	if = {
		limit = {
			OR = {
				has_idea = ALB_albanian_assembly
				has_idea = ALB_administrative_council
				has_idea = ALB_administrative_council_2
			}
		}
		if = {
			limit = { ALB_is_muslim_character = yes }
			custom_trigger_tooltip = {
				tooltip = ALB_can_only_have_one_muslim_advisor_tt
				OR = {
					THIS = { is_political_advisor = yes }
					NOT = {
						any_character = {
							ALB_is_muslim_character = yes
							is_political_advisor = yes
						}
					}
				}
			}
		}
		if = {
			limit = { ALB_is_catholic_character = yes }
			custom_trigger_tooltip = {
				tooltip = ALB_can_only_have_one_catholic_advisor_tt
				OR = {
					THIS = { is_political_advisor = yes }
					NOT = {
						any_character = {
							ALB_is_catholic_character = yes
							is_political_advisor = yes
						}
					}
				}
			}
		}
		if = {
			limit = { ALB_is_orthodox_character = yes }
			custom_trigger_tooltip = {
				tooltip = ALB_can_only_have_one_orthodox_advisor_tt
				OR = {
					THIS = { is_political_advisor = yes }
					NOT = {
						any_character = {
							ALB_is_orthodox_character = yes
							is_political_advisor = yes
						}
					}
				}
			}
		}
	}
	if = {
		limit = {
			OR = {
				has_idea = ALB_general_council
				has_idea = ALB_general_council_2
				has_idea = ALB_general_council_3
			}
		}
		if = {
			limit = {
				OR = {
					ALB_is_orthodox_character = yes
					ALB_is_catholic_character = yes
				}
			}
			custom_trigger_tooltip = {
				tooltip = ALB_cannot_have_catholic_or_orthodox_advisor_tt
				NOT = {
					has_idea = ALB_general_council
					has_idea = ALB_general_council_2
					has_idea = ALB_general_council_3
				}
			}
		}
	}
	if = {
		limit = { has_idea = ALB_christian_council }
		if = {
			limit = { ALB_is_muslim_character = yes }
			custom_trigger_tooltip = {
				tooltip = ALB_cannot_have_muslim_advisor_tt
				NOT = { has_idea = ALB_christian_council }
			}
		}
	}
}
