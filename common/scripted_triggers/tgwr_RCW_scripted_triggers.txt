RCW_core_states_valid_targer = {
	is_controlled_by = ROOT
	NOT = { is_core_of = ROOT }
	OWNER = {
		OR = {
			tag = ROOT
			AND = {
				has_war_with = ROOT
				if = {
					limit = { ROOT = { tag = SOV } }
					is_cossack_tag = no
				}
			}		
		}
	}
}
RCW_recruit_militia_valid_targer = {
	FROM = {
		is_controlled_by = ROOT
		NOT = { has_state_flag = SOV_has_mobilized_militia }
		is_core_of = ROOT
		any_neighbor_state = {
			controller = {
				has_war_with = ROOT
			}
		}
		impassable = no
		is_demilitarized_zone = no
	}
}
