AUS_has_POL_state = {
	custom_trigger_tooltip = {
		any_state = {
			OR = {
				is_owned_and_controlled_by = HUN
				is_owned_and_controlled_by = AUS
			}
			has_state_flag = AUH_POL_state
		}
		tooltip = AUH_POL_state_tt
	}
}
