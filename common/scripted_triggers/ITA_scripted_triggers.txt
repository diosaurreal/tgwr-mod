	OR = {
		original_tag = ITA
		original_tag = SRD
		original_tag = VNC
		original_tag = EMI
		original_tag = PAP
		original_tag = SIC
		original_tag = TUS
		original_tag = FIU
		original_tag = SMR
		original_tag = MLT
	}
}

}

ITA_controls_every_coastal_libyan_state = {
	custom_trigger_tooltip = {
		tooltip = ITA_controls_every_coastal_libyan_state_tt
		all_enemy_country = {
			if = {
				limit = {
					any_owned_state = {
						region = 225
					}
				}
				all_owned_state = {
					if = {
						limit = {
							region = 225
							is_coastal = yes
						}
						ROOT = {
							controls_state = PREV
						}
					}
				}
			}
		}
	}
}

has_italian_civil_war = {
	custom_trigger_tooltip = {
		tooltip = has_italian_civil_war_tt
		OR = {
			ITA = {
				has_civil_war = yes
			}
			any_country = {
				is_italian_nation = yes
				any_enemy_country = {
					is_italian_nation = yes
				}
			}
		}
	}
}
