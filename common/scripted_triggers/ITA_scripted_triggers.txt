is_italian_nation = {
	OR = {
		original_tag = ITA
		original_tag = SRD
		original_tag = VNC
		original_tag = EMI
		original_tag = PAP
		original_tag = SIC
		original_tag = TUS
		original_tag = FIU
		original_tag = SMR
		original_tag = MLT
	}
}

ITA_is_italian_tag_for_music = {
	is_italian_nation = yes
}

has_italian_civil_war = {
	custom_trigger_tooltip = {
		tooltip = has_italian_civil_war_tt
		OR = {
			ITA = {
				has_civil_war = yes
			}
			any_country = {
				is_italian_nation = yes
				any_enemy_country = {
					is_italian_nation = yes
				}
			}
		}
	}
}

ITA_government_is_centrist = {
	custom_trigger_tooltip = {
		tooltip = ITA_gov_center
		OR = {
			has_country_leader = {
				character = ITA_antonio_salandra
				ruling_only = yes
			}
			has_country_leader = {
				character = ITA_romolo_murri
				ruling_only = yes
			}
			has_country_leader = {
				character = ITA_ettore_sacchi
				ruling_only = yes
			}
			has_country_leader = {
				character = ITA_francesco_saverio_nitti
				ruling_only = yes
			}
			has_country_leader = {
				character = ITA_paolo_boselli
				ruling_only = yes
			}
			has_country_leader = {
				character = ITA_vittorio_emanuele_orlando
				ruling_only = yes
			}
			has_country_leader = {
				character = ITA_luigi_luzzatti
				ruling_only = yes
			}
			AND = {
				has_country_leader = {
					character = ITA_giovanni_giolitti
					ruling_only = yes
				}
				NOT = {
					has_global_flag = ITA_national_block_exist
				}
			}
			has_country_leader = {
				character = ITA_sidney_sonnino
				ruling_only = yes
			}
		}
	}
}

ITA_government_is_leftist = {
	custom_trigger_tooltip = {
		tooltip = ITA_gov_left
		OR = {
			has_government = communism
			has_country_leader = { character = ITA_nicola_bombacci }
			has_country_leader = { character = ITA_leonida_bissolati }
		}
	}
}

	}
}
