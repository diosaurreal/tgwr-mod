joint_focus = {
	id = TUR_revive_the_ottomanism_dream
	icon = GFX_TUR_revive_the_ottomanism_dream
	cost = 10
	dynamic = yes
	prerequisite = { focus = TUR_continue_non_partisan_governments focus = TUR_three_pasha_government }
	mutually_exclusive = { focus = TUR_one_language_one_religion_one_nation }
	offset = {
		x = 11
		y = 1
		trigger = { has_focus_tree = ottoman_syria_focus_tree }
	}
	offset = {
		x = 96
		y = 8
		trigger = {
			has_game_rule = {
				rule = obsolete_focus_branches_visibility
				option = HIDE
			}
			has_focus_tree = ottoman_focus_tree
		}
	}
	x = 0
	y = 0
	available = {
		NOT = { has_government = fascism }
		if = {
			limit = { tag = TRS }
			NOT = { has_autonomy_state = autonomy_tgwr_ottoman_committee_government }
		}
	}
	available_if_capitulated = no
	joint_trigger = { TUR_is_committee_government_focus = yes }
	allow_branch = { original_tag = TUR }
	search_filters = {
		FOCUS_FILTER_MINORITY
		FOCUS_FILTER_STABILITY
		FOCUS_FILTER_WAR_SUPPORT
	}
	completion_reward = {
		custom_effect_tooltip = TUR_revive_the_ottomanism_dream_tt
		add_to_variable = { TUR_sunni_population_factor = -0.002 }
		add_to_variable = { TUR_orthodox_population_factor = 0.003 }
		add_to_variable = { TUR_apostolic_population_factor = 0.003 }
		add_to_variable = { TUR_judaic_population_factor = 0.003 }
		add_to_variable = { TUR_syriac_population_factor = 0.003 }
		add_to_variable = { TUR_catholic_population_factor = 0.003 }
		ottoman_minority_variable_check = yes
		add_political_power = -50
	}
}
